1.Vue的安装部署：
* 直接用script进行引用(也可以直接引用一个文件 script)

对于制作原型或学习，你可以这样使用最新版本：
<script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>

对于生产环境，我们推荐链接到一个明确的版本号和构建文件，以避免新版本造成的不可预期的破坏：
<script src="https://cdn.jsdelivr.net/npm/vue@2.6.11"></script>

如果你使用原生 ES Modules，这里也有一个兼容 ES Module 的构建文件：
<script type="module">
      import Vue from 'https://cdn.jsdelivr.net/npm/vue@2.6.11/dist/vue.esm.browser.js'
</script>

3.数据与方法 数据的赋值 以及修改 还有watch的使用

<html>
<head>
    <meta charset="utf-8">
    <title></title>
    <script src="./vue.js" type="text/javascript" charset="UTF-8"></script>
</head>
<body>
    <div id="app">
        {{a}}{{b}}
    </div>

    <script type="text/javascript">
        var data = {a:1,b:2}
        var vm = new Vue({
            el:"#app",
            data:data
        });
    // 修改a的值
    data.a = "text"
    data.b = "lijian"
    // watch方法   a 变量的变化   放在变化之前  观察某一个变量的变化
    vm.$watch('a',function(newValue,oldValue){
        console.log(newValue,oldValue)
    })
    // $的使用  可以通过实例VM来修改
    vm.$data.a = "lijian"
    </script>
</body>
</html>

4.生命周期

<!DOCTYPE html>
<html>
<head>
<meta charset="utf-8">
<title></title>
<script src="vue.js" type="text/javascript" charset="utf-8"></script>
</head>
<body>
<div id="app">
    {{msg}}
</div>
<script type="text/javascript">
var vm = new Vue({
    el : "#app",
    data : {
        msg : "hi vue",
    },
    //在实例初始化之后，数据观测 (data observer) 和 event/watcher 事件配置之前被调用。
    beforeCreate:function(){
        console.log('beforeCreate');
    },
    /* 在实例创建完成后被立即调用。
    在这一步，实例已完成以下的配置：数据观测 (data observer)，属性和方法的运算，watch/event 事件回调。
    然而，挂载阶段还没开始，$el 属性目前不可见。 */
    created    :function(){
        console.log('created');
    },
    //在挂载开始之前被调用：相关的渲染函数首次被调用
    beforeMount : function(){
        console.log('beforeMount');
    },
    //el 被新创建的 vm.$el 替换, 挂在成功    
    mounted : function(){
        console.log('mounted');

    },
    //数据更新时调用
    beforeUpdate : function(){
        console.log('beforeUpdate');

    },
    //组件 DOM 已经更新, 组件更新完毕 
    updated : function(){
        console.log('updated');

    }
});
setTimeout(function(){
    vm.msg = "change ......";
}, 3000);
</script>
</body>
</html>

17.基于axios.js来发送Ajax请求

<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <title></title>
        <!-- vue的远程资源 -->
        <script src="./vue.js" type="text/javascript" charset="UTF-8"></script>
        <!-- axios的远程资源 -->
        <script src="https://cdn.bootcss.com/axios/0.19.2/axios.js"></script>
    </head>
    <body>
        <!-- 加载本地文件 -->
        <div id="box">
            <input type="button" value="加载文件" @click="get_file()" /><br/>
            <input type="button" value="get_url" @click="get_url()"/><br/>
            <input type="button" value="post_url" @click="post_url()"/><br/>
        </div>
        <script type="text/javascript">
            // 定义局部变量
            var data = {};
            // 实例化vuw
            var box = new Vue({
                el:"#box",
                data:data,
                // 实现类函数
                methods:{
                    get_file:function(){
                        console.log("ajax-加载文件");
                        axios.get("./note.md")
                            .then(response =>{
                                console.log(response.data)
                            })
                            .catch(function(error){
                                console.log(error);
                            })
                    },
                    get_url: function(){
                        console.log("get请求");
                        // axios.get("https://api.github.com/search/repositories?q=vue&sort=stars")
                        //  axios.get(url [,config] ) => config -> dict objcet => params传递的参数
                        // axios.get("https://api.github.com/search/repositories", {params: {"q":"vue", "sort":"stars"}})
                        axios.get("http://localhost:5000/check_form?username=abcdefg")
                        // axios.get("http://localhost:5000/check_form", {params:{"username":"abcdefg"}})
                        .then(function(response){
                            console.log(response.status);
                            console.log(response.data);
                        })
                        .catch(function(response){
                            console.log(response);
                        })
                    },
                    post_url: function(){
                        console.log("post请求");
                        axios.post("http://localhost:5000/check_form", {"username":"abcdefg"})
                        .then(function(response){
                            console.log(response.status);
                            console.log(response.data);
                        })
                        .catch(function(response){
                            console.log(response);
                        })
                    }

                }
            })
        </script>
    </body>
</html>
